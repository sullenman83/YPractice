# Через интерфейс добавляется бронь для события, которая записывается в очередь на обработку, а потребителю API  возвращается id бронирования.
Бронирование хранится в памяти. По id брони можно отслеживать статус бронирования. 
Фоновый сервис раз в 5 с получает из очереди на обработку 5 бронирований и обрабатывает их. 

# Модель брониваорния Booking
public class Booking
{
    // Идентификатор брони
    public required Guid Id { get; set; }

    // Идентификатор события, к которому привязана бронь
    public required Guid EventId { get; set; }

    // Текущий статус брони
    public required BookingStatus Status  { get; set; }

    // Дата и время создания брони
    public required DateTime CreatedAt { get; set; }
    
    // Дата и время обработки брони    
    public DateTime? ProcessedAt { get; set; }	
}

#Модель статусов бронирования 
BookingStatus
{    
    // Брони создана, ожидает обработки
    Pending,

    // Бронь подтверждена
    Confirmed,

    // бронь отклонена
    Rejected
}


## Добавлена следующая функциональность 
	# Изменение класса события 
	- Добавлено свойство TotalSeats - общее количество мест
	- Добавлено свойство AvailableSeats - доступное количество мест
	# Изменения класса Booking 
	- Добавлено свойство SeatsCount - количество мест в бронировании 
	# Изменения в API
	- В метод CreateBooking добавлен параметр seatsCount - количество бронируемых мест.
	  Метод CreateBooking теперь может возвращать 409Conflict - если мест недостаточно для бронирования
	
	# Изменения в BookingService
	- Метод CreateBookingAsync теперь безопасен для конкурентного добавления броней
		Синхронизация поток осуществляется при помощи SemaphorSlim
	# Изменения в фоновом сервисе обработки броней BookingHandlerService
	- Оптимизирована производительность. Теперь обработка стала многопоточной, задействуя все аппаратные возможности сервера
		Потокобезопасность обеспечивается при помощи SemaphorSlim (защиьа критической секции в асинхроных методах)
	# Добавлены следующие тесты
		- Тесты для BookingService
			- Успешные сценарии:
				- оздание брони уменьшает AvailableSeats на 1.
				- Создание нескольких броней (до лимита) — все успешны, у каждой уникальный Id
				- После исчерпания мест следующая попытка выбрасывает NoAvailableSeatsException
			- Неуспешные сценарии:
				- Бронирование для несуществующего события → NotFoundException
				- Бронирование при отсутствии мест → NoAvailableSeatsException
		- Тесты на смену статуса брони
			- После вызова Confirm() бронь возвращает статус Confirmed и заполненный ProcessedAt
			- После вызова Reject() бронь возвращает статус Rejected и заполненный ProcessedAt
			- После Reject() ReleaseSeats() количество свободных мест восстанавливается
			- После Reject() ReleaseSeats() можно успешно создать новую бронь на то же место
		- Тесты на конкурентность
			- Тест на защиту от овербукинга				
			- Тест на уникальность Id при конкурентных запросах
